## 6. Идея, метод, алгоритм решения
Составить и отладить простейшую программу на Си
- Напишем простуйшую программу на Си с ошибками для нахождения первых 10 простых чисел 
```c
#include <stdio.h>

int main()
{
    int z, k = 1, i = 0;
    while (i < 10)
    {
        ++k;
        z = 1;
        for (int j = 2; j * j <= k; ++j)
        {
            if (k % j == 0)
            {
                z = 0;
                break;
            }
        }
        if (z == 1)
        {
            printf("%d ", k)
            ++i;
        }
    }
}
```
