## 8. Протокол
#### Напишем программу на Си
```c
#include <stdio.h>


int min(int a, int b) {
    return a > b ? b : a;
}
int max(int a, int b) {
    return a > b ? a : b;
} 
int sign(int a) {
    return a < 0 ? -1 : a == 0 ? 0 : 1;
}
int abs(int a) {
    return a < 0 ? -a : a;
}

typedef struct {
    int i, j, l;
} Point;

int check(Point a) {
    return ((a.i + a.j + 10 <= 0) && (a.i + a.j + 20 >= 0));
}

Point next(Point a, int k) {
    a.i = sign(a.i + 1) * abs(abs(k - a.j) - abs(a.i - a.l));
    a.j = a.j % 20 + max(a.i % 20, min(a.j - k, a.l - k)) - 10;
    a.l = k * (a.i + 1) * (a.j + 2) * (a.l + 3) % 20;
    return a;
}

int main(void)
{
  int pr = 1;
  Point a = {13, 19, 14};
  for (int k = 0; k <= 50; ++k) {
    if (check(a)) {
      printf("YES, k = %d, i = %d, j = %d, l = %d", k, a.i, a.j, a.l);
      pr = 0;
      break;
    } else
      a = next(a, k);
  }
  if (pr == 1)
    printf("NO  i = %d, j = %d, l = %d", a.i, a.j, a.l);
  return 0;
}

```
